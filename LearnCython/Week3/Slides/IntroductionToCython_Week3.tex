\begin{frame}[fragile]{The Sugar}
    sum\_nums.pyx
    \begin{lstlisting}
(*@\colorbox{green!20}{cpdef}@*) unsigned long long sum_nums(unsigned long n):
    cdef unsigned long long s = 0
    cdef unsigned long i
    for i in range(1, n+1):
        s += i
    return s
    \end{lstlisting}
\end{frame}
